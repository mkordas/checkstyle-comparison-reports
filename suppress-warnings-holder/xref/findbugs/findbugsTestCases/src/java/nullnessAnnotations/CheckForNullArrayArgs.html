10  @ParametersAreNonnullByDefault
11  public class CheckForNullArrayArgs {
12  
13      @NoWarning("NP,RCN")
14      protected Object caller(@CheckForNull Object param) {
15          final Object[] paramArray = param == null ? null : new Object[] { param };
16          return methodTakingArray(paramArray);
17      }
18  
19      @NoWarning("NP,RCN")
20      protected Object methodTakingArray(@CheckForNull Object[] params) {
21          return params == null ? Boolean.FALSE : Boolean.TRUE;
22      }
23  
24      @ExpectWarning("IL_INFINITE_RECURSIVE_LOOP")
25      public void infiniteRecursiveLoop() {
26              infiniteRecursiveLoop();
27      }
