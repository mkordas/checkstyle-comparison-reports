5       // Jakarta commons-modeler-1.1
6       // org/apache/commons/modeler/util/IntrospectionUtils.java
7       static boolean same1(int a[], int b[]) {
8           if (a == null)
9               if (b == null || b.length == 0)
10                  return true;
11          if (b == null)
12              if (a == null || a.length == 0)
13                  return true;
14          // Bug is here. If one but not both of a & b are null,
15          // we will get a null pointer exception
16          if (a.length != b.length)
17              return false;
18          return true;
19      }
