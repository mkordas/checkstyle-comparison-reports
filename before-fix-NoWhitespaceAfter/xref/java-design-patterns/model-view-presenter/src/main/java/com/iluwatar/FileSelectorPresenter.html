1   package com.iluwatar;
2   
3   /**
4    * Every instance of this class represents the Presenter component in the
5    * Model-View-Presenter architectural pattern.
6    * 
7    * It is responsible for reacting to the user's actions and update the View
8    * component.
9    */
10  public class FileSelectorPresenter {
11  
12  	/**
13  	 * The View component that the presenter interacts with.
14  	 */
15  	private FileSelectorView view;
16  
17  	/**
18  	 * The Model component that the presenter interacts with.
19  	 */
20  	private FileLoader loader;
21  
22  	/**
23  	 * Constructor
24  	 * 
25  	 * @param view
26  	 *            The view component that the presenter will interact with.
27  	 */
28  	public FileSelectorPresenter(FileSelectorView view) {
29  		this.view = view;
30  	}
31  
32  	/**
33  	 * Sets the FileLoader object, to the value given as parameter.
34  	 * 
35  	 * @param loader
36  	 *            The new FileLoader object(the Model component).
37  	 */
38  	public void setLoader(FileLoader loader) {
39  		this.loader = loader;
40  	}
41  
42  	/**
43  	 * Starts the presenter.
44  	 */
45  	public void start() {
46  		view.setPresenter(this);
47  		view.open();
48  	}
49  
50  	/**
51  	 * An "event" that fires when the name of the file to be loaded changes.
52  	 */
53  	public void fileNameChanged() {
54  		loader.setFileName(view.getFileName());
55  	}
56  
57  	public void confirmed() {
58  		if (loader.getFileName() == null || loader.getFileName().equals("")) {
59  			view.showMessage("Please give the name of the file first!");
60  			return;
61  		}
62  
63  		if (loader.fileExists()) {
64  			String data = loader.loadData();
65  			view.displayData(data);
66  		}
67  
68  		else {
69  			view.showMessage("The file specified does not exist.");
70  		}
71  	}
72  
73  	/**
74  	 * Cancels the file loading process.
75  	 */
76  	public void cancelled() {
77  		view.close();
78  	}
79  }
