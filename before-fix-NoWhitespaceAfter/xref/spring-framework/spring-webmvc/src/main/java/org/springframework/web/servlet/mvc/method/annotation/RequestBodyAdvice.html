17  package org.springframework.web.servlet.mvc.method.annotation;
18  
19  import java.io.IOException;
20  import java.lang.reflect.Type;
21  
22  import org.springframework.core.MethodParameter;
23  import org.springframework.http.HttpInputMessage;
24  import org.springframework.http.converter.HttpMessageConverter;
25  
26  
27  /**
28   * Allows customizing the request before its body is read and converted into an
29   * Object and also allows the resulting Object before it is passed into a
30   * controller method as an {code @RequestBody} or an {@code HttpEntity} method
31   * argument.
32   *
33   * <p>Implementations of this contract may be registered directly with the
34   * {@code RequestMappingHandlerAdapter} or more likely annotated with
35   * {@code @ControllerAdvice} in which case they are auto-detected.
36   *
37   * @author Rossen Stoyanchev
38   * @since 4.2
39   */
40  public interface RequestBodyAdvice {
41  
42  	/**
43  	 * Invoked first to determine if this interceptor applies.
44  	 * @param methodParameter the method parameter
45  	 * @param targetType the target type, not necessarily the same as the method
46  	 * parameter type, e.g. for {@code HttpEntity<String>}.
47  	 * @param converterType the selected converter type
48  	 * @return whether this interceptor should be invoked or not
49  	 */
50  	boolean supports(MethodParameter methodParameter, Type targetType,
51  			Class<? extends HttpMessageConverter<?>> converterType);
52  
53  	/**
54  	 * Invoked second (and last) if the body is empty.
55  	 * @param body set to {@code null} before the first advice is called
56  	 * @param inputMessage the request
57  	 * @param parameter the method parameter
58  	 * @param targetType the target type, not necessarily the same as the method
59  	 * parameter type, e.g. for {@code HttpEntity<String>}.
60  	 * @param converterType the selected converter type
61  	 * @return the value to use or {@code null} which may then raise an
62  	 * {@code HttpMessageNotReadableException} if the argument is required.
63  	 */
64  	Object handleEmptyBody(Object body, HttpInputMessage inputMessage, MethodParameter parameter,
65  			Type targetType, Class<? extends HttpMessageConverter<?>> converterType);
66  
67  	/**
68  	 * Invoked second before the request body is read and converted.
69  	 * @param inputMessage the request
70  	 * @param parameter the target method parameter
71  	 * @param targetType the target type, not necessarily the same as the method
72  	 * parameter type, e.g. for {@code HttpEntity<String>}.
73  	 * @param converterType the converter used to deserialize the body
74  	 * @return the input request or a new instance, never {@code null}
75  	 */
76  	HttpInputMessage beforeBodyRead(HttpInputMessage inputMessage, MethodParameter parameter,
77  			Type targetType, Class<? extends HttpMessageConverter<?>> converterType) throws IOException;
78  
79  	/**
80  	 * Invoked third (and last) after the request body is converted to an Object.
81  	 * @param body set to the converter Object before the 1st advice is called
82  	 * @param inputMessage the request
83  	 * @param parameter the target method parameter
84  	 * @param targetType the target type, not necessarily the same as the method
85  	 * parameter type, e.g. for {@code HttpEntity<String>}.
86  	 * @param converterType the converter used to deserialize the body
87  	 * @return the same body or a new instance
88  	 */
89  	Object afterBodyRead(Object body, HttpInputMessage inputMessage, MethodParameter parameter,
90  			Type targetType, Class<? extends HttpMessageConverter<?>> converterType);
91  
92  }
