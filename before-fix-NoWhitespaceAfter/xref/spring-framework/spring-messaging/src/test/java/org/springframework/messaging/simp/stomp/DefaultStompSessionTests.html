184 		verify(this.connection).send(any());
185 
186 		this.connectHeaders.setHeartbeat(new long[] {10000, 10000});
187 
188 		StompHeaderAccessor connected = StompHeaderAccessor.create(StompCommand.CONNECTED);
189 		connected.setVersion("1.2");
190 		connected.setHeartbeat(10000, 10000);
191 		connected.setLeaveMutable(true);
192 		this.session.handleMessage(MessageBuilder.createMessage(new byte[0], connected.getMessageHeaders()));
193 
194 		ArgumentCaptor<Runnable> writeTaskCaptor = ArgumentCaptor.forClass(Runnable.class);
195 		ArgumentCaptor<Runnable> readTaskCaptor = ArgumentCaptor.forClass(Runnable.class);
196 		verify(this.connection).onWriteInactivity(writeTaskCaptor.capture(), any(Long.class));
197 		verify(this.connection).onReadInactivity(readTaskCaptor.capture(), any(Long.class));
198 
199 		Runnable writeTask = writeTaskCaptor.getValue();
200 		Runnable readTask = readTaskCaptor.getValue();
201 		assertNotNull(writeTask);
202 		assertNotNull(readTask);
203 
204 		writeTask.run();
205 		StompHeaderAccessor accessor = StompHeaderAccessor.createForHeartbeat();
206 		Message<byte[]> message = MessageBuilder.createMessage(new byte[] {'\n'}, accessor.getMessageHeaders());
207 		verify(this.connection).send(eq(message));
208 		verifyNoMoreInteractions(this.connection);
209 
210 		reset(this.sessionHandler);
211 		readTask.run();
212 		verify(this.sessionHandler).handleTransportError(same(this.session), any(IllegalStateException.class));
213 		verifyNoMoreInteractions(this.sessionHandler);
214 	}

542 	@SuppressWarnings({ "unchecked", "rawtypes" })
543 	public void receiptNotReceived() throws Exception {
544 
545 		TaskScheduler taskScheduler = mock(TaskScheduler.class);
546 
547 		this.session.afterConnected(this.connection);
548 		this.session.setTaskScheduler(taskScheduler);
549 
550 		AtomicReference<Boolean> notReceived = new AtomicReference<>();
551 
552 		ScheduledFuture future = mock(ScheduledFuture.class);
553 		when(taskScheduler.schedule(any(Runnable.class), any(Date.class))).thenReturn(future);
554 
555 		StompHeaders headers = new StompHeaders();
556 		headers.setDestination("/topic/foo");
557 		headers.setReceipt("my-receipt");
558 		Receiptable receiptable = this.session.send(headers, "payload");
559 		receiptable.addReceiptLostTask(() -> notReceived.set(true));
560 
561 		ArgumentCaptor<Runnable> taskCaptor = ArgumentCaptor.forClass(Runnable.class);
562 		verify(taskScheduler).schedule(taskCaptor.capture(), notNull(Date.class));
563 		Runnable scheduledTask = taskCaptor.getValue();
564 		assertNotNull(scheduledTask);
565 
566 		assertNull(notReceived.get());
567 
568 		scheduledTask.run();
569 		assertTrue(notReceived.get());
570 		verify(future).cancel(true);
571 		verifyNoMoreInteractions(future);
572 	}
