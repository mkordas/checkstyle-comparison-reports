16  package org.springframework.messaging.handler;
17  
18  import org.springframework.core.Ordered;
19  
20  /**
21   * Represents a Spring-managed bean with cross-cutting functionality to be
22   * applied to one or more Spring beans with annotation-based message
23   * handling methods.
24   *
25   * <p>Component stereotypes such as
26   * {@link org.springframework.stereotype.Controller @Controller} with annotation
27   * handler methods often need cross-cutting functionality across all or a subset
28   * of such annotated components. A primary example of this is the need for "global"
29   * annotated exception handler methods but the concept applies more generally.
30   *
31   * @author Rossen Stoyanchev
32   * @since 4.2
33   */
34  public interface MessagingAdviceBean extends Ordered {
35  
36  	/**
37  	 * Return the type of the contained advice bean.
38  	 * <p>If the bean type is a CGLIB-generated class, the original user-defined
39  	 * class is returned.
40  	 */
41  	Class<?> getBeanType();
42  
43  	/**
44  	 * Return the advice bean instance, if necessary resolving a bean specified
45  	 * by name through the BeanFactory.
46  	 */
47  	Object resolveBean();
48  
49  	/**
50  	 * Whether this {@link MessagingAdviceBean} applies to the given bean type.
51  	 * @param beanType the type of the bean to check
52  	 */
53  	boolean isApplicableToBeanType(Class<?> beanType);
54  
55  }
