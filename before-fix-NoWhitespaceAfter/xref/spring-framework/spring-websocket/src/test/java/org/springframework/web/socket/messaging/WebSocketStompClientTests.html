130 		String text = "SEND\na:alpha\n\nMessage payload\0";
131 		connect().handleMessage(this.webSocketSession, new TextMessage(text));
132 
133 		ArgumentCaptor<Message> captor = ArgumentCaptor.forClass(Message.class);

136 		assertNotNull(message);
137 
138 		StompHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);
139 		StompHeaders headers = StompHeaders.readOnlyStompHeaders(accessor.toNativeHeaderMap());
140 		assertEquals(StompCommand.SEND, accessor.getCommand());
141 		assertEquals("alpha", headers.getFirst("a"));
142 		assertEquals("Message payload", new String(message.getPayload(), UTF_8));
143 	}

304 
305 		this.webSocketHandlerCaptor.getValue().afterConnectionClosed(this.webSocketSession, CloseStatus.NORMAL);

331 	private void testInactivityTaskScheduling(Runnable runnable, long delay, long sleepTime)
332 			throws InterruptedException {
333 
334 		ArgumentCaptor<Runnable> inactivityTaskCaptor = ArgumentCaptor.forClass(Runnable.class);
335 		verify(this.taskScheduler).scheduleWithFixedDelay(inactivityTaskCaptor.capture(), eq(delay/2));
336 		verifyNoMoreInteractions(this.taskScheduler);
337 
338 		if (sleepTime > 0) {
339 			Thread.sleep(sleepTime);
340 		}
341 
342 		Runnable inactivityTask = inactivityTaskCaptor.getValue();
343 		assertNotNull(inactivityTask);
344 		inactivityTask.run();
345 
346 		if (sleepTime > 0) {
347 			verify(runnable).run();
348 		}
349 		else {
350 			verifyNoMoreInteractions(runnable);
351 		}
352 	}
