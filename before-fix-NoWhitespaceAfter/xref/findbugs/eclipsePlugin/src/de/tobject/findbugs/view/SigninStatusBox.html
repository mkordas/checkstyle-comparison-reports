30  public class SigninStatusBox extends Composite {
31      private final CloudStatusListener cloudStatusListener;
32  
33      private Cloud cloud;
34  
35      private Label statusLine;
36  
37      public SigninStatusBox(Composite parent, int style) {
38          super(parent, style);
39  
40          cloudStatusListener = new CloudStatusListener() {
41              public void handleStateChange(SigninState oldState, SigninState state) {
42                  updateLabel();
43              }
44  
45              public void handleIssueDataDownloadedEvent() { // ok
46              }
47          };
48          createControls();
49          parent.addDisposeListener(new DisposeListener() {
50              public void widgetDisposed(DisposeEvent e) {
51                  dispose();
52              }
53          });
54          addControlListener(new ControlListener() {
55  
56              public void controlResized(ControlEvent e) {
57                  statusLine.pack(true);
58                  SigninStatusBox.this.pack(true);
59              }
60  
61              public void controlMoved(ControlEvent e) {
62                  // noop
63              }
64          });
65      }
66  
67      @Override
68      public void dispose() {
69          if (statusLine.isDisposed()) {
70              return;
71          }
72          statusLine.getCursor().dispose();
73          super.dispose();
74          if (this.cloud != null) {
75              this.cloud.removeStatusListener(cloudStatusListener);
76          }
77      }
78  
79      public void setCloud(Cloud cloud) {
80          if (this.cloud != null) {
81              this.cloud.removeStatusListener(cloudStatusListener);
82          }
83          this.cloud = cloud;
84          if (cloud != null) {
85              cloud.addStatusListener(cloudStatusListener);
86          }
87          updateLabel();
88      }
89  
90      private void createControls() {
91          this.setLayout(new GridLayout(1, true));
92          statusLine = new Label(this, SWT.NONE);
93          statusLine.setLayoutData(new GridData(SWT.RIGHT, GridData.CENTER, true, false));
94          statusLine.setAlignment(SWT.CENTER);
95          statusLine.setCursor(new Cursor(null, SWT.CURSOR_HAND));
96          statusLine.addMouseListener(new MouseAdapter() {
97              @Override
98              public void mouseDown(MouseEvent e) {
99                  showSigninPopupMenu(statusLine.toDisplay(e.x, e.y));
100             }
101         });
102         updateLabel();
103     }
104 
105     private void showSigninPopupMenu(Point p) {
106         Menu popupMenu = new Menu(this);
107         MenuItem automaticCheckbox = new MenuItem(popupMenu, SWT.CHECK);
108         automaticCheckbox.setText("Sign in automatically");
109         SigninState state = cloud.getSigninState();
110         automaticCheckbox.setEnabled(state != SigninState.NO_SIGNIN_REQUIRED);
111         final boolean origSelection = WebCloudNameLookup.isSavingSessionInfoEnabled();
112         automaticCheckbox.setSelection(origSelection);
113         automaticCheckbox.addSelectionListener(new SelectionAdapter() {
114             @Override
115             public void widgetSelected(SelectionEvent e) {
116                 WebCloudNameLookup.setSaveSessionInformation(!origSelection);
117             }
118         });
119 
120         if (state.couldSignIn()) {
121             MenuItem signInItem = new MenuItem(popupMenu, SWT.NONE);
122             signInItem.setText("Sign in");
123             signInItem.addSelectionListener(new SelectionAdapter() {
124                 @Override
125                 public void widgetSelected(SelectionEvent e) {
126                     try {
127                         cloud.signIn();
128                     } catch (IOException e1) {
129                         MessageDialog.openError(FindbugsPlugin.getShell(), "Error", e1.toString());
130                     }
131                 }
132             });
133 
134         } else if (state == SigninState.SIGNING_IN || state == SigninState.SIGNED_IN) {
135             MenuItem signOutItem = new MenuItem(popupMenu, SWT.NONE);
136             signOutItem.setText("Sign out");
137             signOutItem.addSelectionListener(new SelectionAdapter() {
138                 @Override
139                 public void widgetSelected(SelectionEvent e) {
140                     cloud.signOut();
141                 }
142             });
143         }
144         popupMenu.setLocation(p);
145         popupMenu.setVisible(true);
146     }
147 
148     private void updateLabel() {
149         FindbugsPlugin.getShell().getDisplay().asyncExec(new Runnable() {
150             public void run() {
151                 if (statusLine != null && !statusLine.isDisposed()) {
152                     if (cloud != null) {
153                         String text = cloud.getCloudName() + "\n" + cloud.getSigninState();
154                         if (cloud.getUser() != null) {
155                             text += ": " + cloud.getUser();
156                         }
157                         statusLine.setText(text);
158                         statusLine.pack(true);
159                         SigninStatusBox.this.pack(true);
160                     } else {
161                         statusLine.setText("");
162                     }
163                 }
164             }
165         });
166     }
