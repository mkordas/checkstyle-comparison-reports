20  package edu.umd.cs.findbugs.detect;
21  
22  import org.apache.bcel.classfile.Code;
23  
24  import edu.umd.cs.findbugs.BugInstance;
25  import edu.umd.cs.findbugs.BugReporter;
26  import edu.umd.cs.findbugs.OpcodeStack.Item;
27  import edu.umd.cs.findbugs.ba.ClassContext;
28  import edu.umd.cs.findbugs.ba.XField;
29  import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
30  
31  /**
32   * @author Tagir Valeev
33   */
34  public class MutableEnum extends OpcodeStackDetector {
35  
36      private final BugReporter reporter;
37      private boolean skip;
38  
39      public MutableEnum(BugReporter reporter) {
40          this.reporter = reporter;
41      }
42  
43      @Override
44      public void visitClassContext(ClassContext classContext) {
45          if(!classContext.getJavaClass().isEnum() || !classContext.getJavaClass().isPublic()) {
46              return;
47          }
48          boolean hasInterestingField = false;
49          for(XField field : classContext.getXClass().getXFields()) {
50              if(!field.isStatic() && !field.isFinal() && !field.isSynthetic()) {
51                  if(field.isPublic()) {
52                      reporter.reportBug(new BugInstance("ME_MUTABLE_ENUM_FIELD", NORMAL_PRIORITY).addClass(classContext.getJavaClass())
53                              .addField(field));
54                  } else {
55                      hasInterestingField = true;
56                  }
57              }
58          }
59          if(hasInterestingField) {
60              super.visitClassContext(classContext);
61          }
62      }
63  
64      @Override
65      public boolean shouldVisitCode(Code obj) {
66          skip = false;
67          if(getXMethod().isPublic() && getNumberMethodArguments() > 0) {
68              return true;
69          }
70          return false;
71      }
72  
73      @Override
74      public void sawOpcode(int seen) {
75          if(skip) {
76              return;
77          }
78          if(isBranch(seen) || seen == ATHROW || isReturn(seen)) {
79              skip = true;
80          }
81          if(seen == PUTFIELD) {
82              XField xField = getXFieldOperand();
83              if(xField != null && xField.getClassDescriptor().getClassName().equals(getClassName())) {
84                  Item val = getStack().getStackItem(0);
85                  if(val.isInitialParameter()) {
86                      reporter.reportBug(new BugInstance("ME_ENUM_FIELD_SETTER", NORMAL_PRIORITY).addClassAndMethod(this).addField(xField)
87                              .addSourceLine(this));
88                  }
89              }
90          }
91      }
92  }
