38  /**
39   * Analysis engine to produce TypeDataflow objects for analyzed methods.
40   *
41   * @author David Hovemeyer
42   */
43  public class TypeDataflowFactory extends AnalysisFactory<TypeDataflow> {
44      /**
45       * Constructor.
46       */
47      public TypeDataflowFactory() {
48          super("type analysis", TypeDataflow.class);
49      }
50  
51      /*
52       * (non-Javadoc)
53       *
54       * @see
55       * edu.umd.cs.findbugs.classfile.IAnalysisEngine#analyze(edu.umd.cs.findbugs
56       * .classfile.IAnalysisCache, java.lang.Object)
57       */
58      @Override
59      public TypeDataflow analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor) throws CheckedAnalysisException {
60          MethodGen methodGen = getMethodGen(analysisCache, descriptor);
61          if (methodGen == null) {
62              throw new MethodUnprofitableException(descriptor);
63          }
64          CFG cfg = getCFG(analysisCache, descriptor);
65          DepthFirstSearch dfs = getDepthFirstSearch(analysisCache, descriptor);
66          ExceptionSetFactory exceptionSetFactory = getExceptionSetFactory(analysisCache, descriptor);
67          Method method = getMethod(analysisCache, descriptor);
68  
69          TypeAnalysis typeAnalysis = new TypeAnalysis(method, methodGen, cfg, dfs, AnalysisContext.currentAnalysisContext()
70                  .getLookupFailureCallback(), exceptionSetFactory);
71  
72          if (AnalysisContext.currentAnalysisContext().getBoolProperty(AnalysisFeatures.MODEL_INSTANCEOF)) {
73              typeAnalysis.setValueNumberDataflow(getValueNumberDataflow(analysisCache, descriptor));
74          }
75  
76          // Field store type database.
77          // If present, this can give us more accurate type information
78          // for values loaded from fields.
79          typeAnalysis.setFieldStoreTypeDatabase(AnalysisContext.currentAnalysisContext().getFieldStoreTypeDatabase());
80  
81          TypeDataflow typeDataflow = new TypeDataflow(cfg, typeAnalysis);
82          try {
83              typeDataflow.execute();
84          } catch (CheckedAnalysisException e) {
85              AnalysisContext.logError("Error performing type dataflow analysis of " + descriptor, e);
86              throw e;
87          }
88          if (TypeAnalysis.DEBUG || ClassContext.DUMP_DATAFLOW_ANALYSIS) {
89              ClassContext.dumpTypeDataflow(method, cfg, typeDataflow);
90          }
91  
92          return typeDataflow;
93      }
94  }
