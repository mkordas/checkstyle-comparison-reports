20  package edu.umd.cs.findbugs.classfile.engine.bcel;
21  
22  import org.apache.bcel.classfile.JavaClass;
23  import org.apache.bcel.classfile.Method;
24  
25  import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
26  import edu.umd.cs.findbugs.classfile.DescriptorFactory;
27  import edu.umd.cs.findbugs.classfile.IAnalysisCache;
28  import edu.umd.cs.findbugs.classfile.MethodDescriptor;
29  
30  /**
31   * Method analysis engine to produce BCEL Method objects.
32   *
33   * @author David Hovemeyer
34   */
35  public class MethodFactory extends AnalysisFactory<Method> {
36  
37      public MethodFactory() {
38          super("Method factory", Method.class);
39      }
40  
41      /*
42       * (non-Javadoc)
43       *
44       * @see
45       * edu.umd.cs.findbugs.classfile.IAnalysisEngine#analyze(edu.umd.cs.findbugs
46       * .classfile.IAnalysisCache, java.lang.Object)
47       */
48      @Override
49      public Method analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor) throws CheckedAnalysisException {
50          JavaClass jclass = analysisCache.getClassAnalysis(JavaClass.class, descriptor.getClassDescriptor());
51          Method[] methodList = jclass.getMethods();
52  
53          Method result = null;
54  
55          // As a side-effect, cache all of the Methods for this JavaClass
56          for (Method method : methodList) {
57              MethodDescriptor methodDescriptor = DescriptorFactory.instance().getMethodDescriptor(
58                      descriptor.getSlashedClassName(), method.getName(), method.getSignature(), method.isStatic());
59  
60              // Put in cache eagerly
61              analysisCache.eagerlyPutMethodAnalysis(Method.class, methodDescriptor, method);
62  
63              if (methodDescriptor.equals(descriptor)) {
64                  result = method;
65              }
66          }
67  
68          return result;
69      }
70  
71      /*
72       * (non-Javadoc)
73       *
74       * @see
75       * edu.umd.cs.findbugs.classfile.IAnalysisEngine#registerWith(edu.umd.cs
76       * .findbugs.classfile.IAnalysisCache)
77       */
78      @Override
79      public void registerWith(IAnalysisCache analysisCache) {
80          analysisCache.registerMethodAnalysisEngine(Method.class, this);
81      }
82  
83  }
