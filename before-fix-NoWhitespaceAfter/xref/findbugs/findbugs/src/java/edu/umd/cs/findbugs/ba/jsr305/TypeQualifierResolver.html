37  /**
38   * Resolve annotations into type qualifiers.
39   *
40   * @author William Pugh
41   */
42  public class TypeQualifierResolver {
43      static final ClassDescriptor typeQualifier = DescriptorFactory.createClassDescriptor(javax.annotation.meta.TypeQualifier.class);
44  
45      static final ClassDescriptor typeQualifierNickname = DescriptorFactory
46              .createClassDescriptor(javax.annotation.meta.TypeQualifierNickname.class);
47  
48      static final ClassDescriptor typeQualifierDefault = DescriptorFactory
49              .createClassDescriptor(javax.annotation.meta.TypeQualifierDefault.class);
50  
51      static final ClassDescriptor elementTypeDescriptor = DescriptorFactory
52              .createClassDescriptor(java.lang.annotation.ElementType.class);
53  
54      static final ClassDescriptor googleNullable = DescriptorFactory.createClassDescriptor("com/google/common/base/Nullable");
55  
56      static final ClassDescriptor intellijNullable = DescriptorFactory.createClassDescriptor("org/jetbrains/annotations/Nullable");
57  
58      static final ClassDescriptor intellijNotNull = DescriptorFactory.createClassDescriptor("org/jetbrains/annotations/NotNull");
59  
60      static final ClassDescriptor eclipseNullable = DescriptorFactory.createClassDescriptor("org/eclipse/jdt/annotation/Nullable");
61  
62      static final ClassDescriptor eclipseNonNull = DescriptorFactory.createClassDescriptor("org/eclipse/jdt/annotation/NonNull");
63  
64      // javax.annotations.ParametersAreNonnullByDefault ?
65      static final ClassDescriptor eclipseNonNullByDefault = DescriptorFactory.createClassDescriptor("org/eclipse/jdt/annotation/NonNullByDefault");
66  
67      /**
68       * Resolve an AnnotationValue into a list of AnnotationValues representing
69       * type qualifier annotations.
70       *
71       * @param value
72       *            AnnotationValue representing the use of an annotation
73       * @return Collection of AnnotationValues representing resolved
74       *         TypeQualifier annotations
75       */
76      public static Collection<AnnotationValue> resolveTypeQualifiers(AnnotationValue value) {
77          LinkedList<AnnotationValue> result = new LinkedList<AnnotationValue>();
78          resolveTypeQualifierNicknames(value, result, new LinkedList<ClassDescriptor>());
79          return result;
80      }
81  
82      /**
83       * Resolve collection of AnnotationValues (which have been used to annotate
84       * an AnnotatedObject or method parameter) into collection of resolved type
85       * qualifier AnnotationValues.
86       *
87       * @param values
88       *            Collection of AnnotationValues used to annotate an
89       *            AnnotatedObject or method parameter
90       * @return Collection of resolved type qualifier AnnotationValues
91       */
92      public static Collection<AnnotationValue> resolveTypeQualifierDefaults(Collection<AnnotationValue> values,
93              ElementType elementType) {
94          LinkedList<AnnotationValue> result = new LinkedList<AnnotationValue>();
95          for (AnnotationValue value : values) {
96              resolveTypeQualifierDefaults(value, elementType, result);
97          }
98          return result;
99      }

101     /**
102      * Resolve an annotation into AnnotationValues representing any type
103      * qualifier(s) the annotation resolves to. Detects annotations which are
104      * directly marked as TypeQualifier annotations, and also resolves the use
105      * of TypeQualifierNickname annotations.
106      *
107      * @param value
108      *            AnnotationValue representing the use of an annotation
109      * @param result
110      *            LinkedList containing resolved type qualifier AnnotationValues
111      * @param onStack
112      *            stack of annotations being processed; used to detect cycles in
113      *            type qualifier nicknames
114      */
115     private static void resolveTypeQualifierNicknames(AnnotationValue value, LinkedList<AnnotationValue> result,
116             LinkedList<ClassDescriptor> onStack) {
117         ClassDescriptor annotationClass = value.getAnnotationClass();
118 
119         if (onStack.contains(annotationClass)) {
120             AnalysisContext.logError("Cycle found in type nicknames: " + onStack);
121             return;
122         }
123         try {
124             onStack.add(annotationClass);
125 
126             try {
127                 if (annotationClass.equals(googleNullable)
128                         || annotationClass.equals(eclipseNullable)
129                         || annotationClass.equals(intellijNullable)) {
130                     resolveTypeQualifierNicknames(new AnnotationValue(JSR305NullnessAnnotations.CHECK_FOR_NULL), result, onStack);
131                     return;
132                 }
133                 if (annotationClass.equals(eclipseNonNull) || annotationClass.equals(eclipseNonNullByDefault)
134                         || annotationClass.equals(intellijNotNull)) {
135                     resolveTypeQualifierNicknames(new AnnotationValue(JSR305NullnessAnnotations.NONNULL), result, onStack);
136                     return;
137                 }
138 
139                 XClass c = Global.getAnalysisCache().getClassAnalysis(XClass.class, annotationClass);
140                 if (c.getAnnotationDescriptors().contains(typeQualifierNickname)) {
141                     for (ClassDescriptor d : c.getAnnotationDescriptors()) {
142                         if (!d.equals(typeQualifierNickname)) {
143                             resolveTypeQualifierNicknames(c.getAnnotation(d), result, onStack);
144                         }
145                     }
146                 } else if (c.getAnnotationDescriptors().contains(typeQualifier)) {
147                     result.add(value);
148                 }
149             } catch (MissingClassException e) {
150                 logMissingAnnotationClass(e);
151                 return;
152             } catch (CheckedAnalysisException e) {
153                 AnalysisContext.logError("Error resolving " + annotationClass, e);
154                 return;
155             }
156 
157         } finally {
158             onStack.removeLast();
159         }
160 
161     }
162 
163     public static void logMissingAnnotationClass(MissingClassException e) {
164         ClassDescriptor c = e.getClassDescriptor();
165         if (c.getClassName().startsWith("javax.annotation")) {
166             AnalysisContext.currentAnalysisContext().getLookupFailureCallback().reportMissingClass(c);
167         }
168     }
169 
170     /**
171      * Resolve collection of AnnotationValues (which have been used to annotate
172      * an AnnotatedObject or method parameter) into collection of resolved type
173      * qualifier AnnotationValues.
174      *
175      * @param values
176      *            Collection of AnnotationValues used to annotate an
177      *            AnnotatedObject or method parameter
178      * @return Collection of resolved type qualifier AnnotationValues
179      */
180     public static Collection<AnnotationValue> resolveTypeQualifiers(Collection<AnnotationValue> values) {
181         if (values.isEmpty()) {
182             return Collections.emptyList();
183         }
184         LinkedList<AnnotationValue> result = new LinkedList<AnnotationValue>();
185         LinkedList<ClassDescriptor> onStack = new LinkedList<ClassDescriptor>();
186         for (AnnotationValue value : values) {
187             resolveTypeQualifierNicknames(value, result, onStack);
188         }
189         return result;
190     }
191 
192     /**
193      * Resolve an annotation into AnnotationValues representing any type
194      * qualifier(s) the annotation resolves to. Detects annotations which are
195      * directly marked as TypeQualifier annotations, and also resolves the use
196      * of TypeQualifierNickname annotations.
197      *
198      * @param value
199      *            AnnotationValue representing the use of an annotation
200      * @param result
201      *            LinkedList containing resolved type qualifier AnnotationValues
202      * @param onStack
203      *            stack of annotations being processed; used to detect cycles in
204      *            type qualifier nicknames
205      */
206     private static void resolveTypeQualifierDefaults(AnnotationValue value, ElementType defaultFor,
207             LinkedList<AnnotationValue> result) {
208 
209         try {
210             XClass c = Global.getAnalysisCache().getClassAnalysis(XClass.class, value.getAnnotationClass());
211             AnnotationValue defaultAnnotation = c.getAnnotation(typeQualifierDefault);
212             if (defaultAnnotation == null) {
213                 return;
214             }
215             for (Object o : (Object[]) defaultAnnotation.getValue("value")) {
216                 if (o instanceof EnumValue) {
217                     EnumValue e = (EnumValue) o;
218                     if (e.desc.equals(elementTypeDescriptor) && e.value.equals(defaultFor.name())) {
219                         for (ClassDescriptor d : c.getAnnotationDescriptors()) {
220                             if (!d.equals(typeQualifierDefault)) {
221                                 resolveTypeQualifierNicknames(c.getAnnotation(d), result, new LinkedList<ClassDescriptor>());
222                             }
223                         }
224                         break;
225                     }
226                 }
227             }
228 
229         } catch (MissingClassException e) {
230             logMissingAnnotationClass(e);
231         } catch (CheckedAnalysisException e) {
232             AnalysisContext.logError("Error resolving " + value.getAnnotationClass(), e);
233 
234         } catch (ClassCastException e) {
235             AnalysisContext.logError("ClassCastException " + value.getAnnotationClass(), e);
236 
237         }
238 
239     }
