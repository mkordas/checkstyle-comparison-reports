20  package edu.umd.cs.findbugs.gui2;
21  
22  import java.io.File;
23  
24  import edu.umd.cs.findbugs.util.Util;
25  
26  enum SaveType {
27      NOT_KNOWN, HTML_OUTPUT, XML_ANALYSIS, FBP_FILE, FBA_FILE;
28      public FindBugsFileFilter getFilter() {
29          switch (this) {
30          case XML_ANALYSIS:
31              return FindBugsAnalysisFileFilter.INSTANCE;
32          case HTML_OUTPUT:
33              return FindBugsHtmlFileFilter.INSTANCE;
34          case FBP_FILE:
35              return FindBugsFBPFileFilter.INSTANCE;
36          case FBA_FILE:
37              return FindBugsFBAFileFilter.INSTANCE;
38          default:
39              throw new IllegalArgumentException("No filter for type NOT_UNKNOWN");
40          }
41      }
42  
43      public boolean isValid(File f) {
44  
45          if (f.isDirectory()) {
46              return false;
47          }
48          FindBugsFileFilter filter = getFilter();
49          return filter.accept(f);
50      }
51  
52      public String getFileExtension() {
53          switch (this) {
54  
55          case HTML_OUTPUT:
56              return ".html";
57          case XML_ANALYSIS:
58              return ".xml";
59          case FBP_FILE:
60              return ".fbp";
61          case FBA_FILE:
62              return ".fba";
63          default:
64              throw new IllegalArgumentException("No filter for type NOT_UNKNOWN");
65          }
66      }
67  
68      public static SaveType forFile(File f) {
69          String extension = Util.getFileExtension(f);
70  
71          if ("html".equals(extension) || "htm".equals(extension)) {
72              return HTML_OUTPUT;
73          }
74          if ("fba".equals(extension)) {
75              return FBA_FILE;
76          }
77          if ("fbp".equals(extension)) {
78              return FBP_FILE;
79          }
80          if ("xml".equals(extension)) {
81              return XML_ANALYSIS;
82          }
83          if ("html".equals(extension)) {
84              return XML_ANALYSIS;
85          }
86          if (f.getName().toLowerCase().endsWith("xml.gz")) {
87              return XML_ANALYSIS;
88          }
89          return NOT_KNOWN;
90      }
91  }
