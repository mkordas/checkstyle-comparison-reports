3   import java.io.File;
4   import java.io.FileOutputStream;
5   import java.io.IOException;
6   import java.io.PrintWriter;
7   import javax.swing.JOptionPane;
8   
9   /**
10   * Used to log any exceptions that are thrown. Can be used to log or debug.
11   * If debugging can use to pop a JOptionPane with the stackTrace.
12   * @author Kristin Stephens
13   *
14   */
15  public class Log
16  {
17      //If true will pop a window with the stackTrace in it.
18      //If false will simply log onto logFile.
19      private final static boolean DEBUG = true;
20      private static File logFile;
21  
22      /**
23       * Set where Log will log any exceptions.
24       * @param logFile
25       */
26      static void setPath(File logFile)
27      {
28          Log.logFile = logFile;
29      }
30  
31      /**
32       * Takes the exception and dependi on DEBUG either posts onto a dialog box
33       * or will record it for later use.
34       *
35       * @param e
36       */
37      static void recordBug(Exception e)
38      {
39          StringBuffer msg = new StringBuffer();
40          msg.append(e.getClass().getName() + ": " + e.getMessage() + "\n");
41          for (StackTraceElement i : e.getStackTrace())
42              msg.append(i + "\n");
43  
44          if (DEBUG)
45              JOptionPane.showMessageDialog(null, msg);
46          else
47          {
48              try
49              {
50                  logFile.createNewFile();
51                  PrintWriter pw = new PrintWriter(new FileOutputStream(logFile));
52                  pw.println(msg);
53                  pw.close();
54              }
55              catch (IOException ohWell) {}
56          }
57      }
58  }
