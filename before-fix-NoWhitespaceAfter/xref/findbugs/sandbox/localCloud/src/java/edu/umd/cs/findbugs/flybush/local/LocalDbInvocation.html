3   import javax.jdo.annotations.IdGeneratorStrategy;
4   import javax.jdo.annotations.IdentityType;
5   import javax.jdo.annotations.PersistenceCapable;
6   import javax.jdo.annotations.Persistent;
7   import javax.jdo.annotations.PrimaryKey;
8   
9   import edu.umd.cs.findbugs.flybush.DbInvocation;
10  import edu.umd.cs.findbugs.flybush.DbUser;
11  
12  @PersistenceCapable(identityType = IdentityType.APPLICATION)
13  public class LocalDbInvocation implements DbInvocation {
14      @SuppressWarnings({"UnusedDeclaration"})
15      @PrimaryKey
16      @Persistent(valueStrategy = IdGeneratorStrategy.IDENTITY)
17      private Long key;
18  
19      @Persistent private LocalDbUser who;
20      @Persistent protected long startTime;
21      @Persistent protected long endTime;
22  
23      public DbUser getWho() {
24          return who;
25      }
26  
27      public void setWho(Object user) {
28          who = (LocalDbUser) user;
29      }
30  
31      public long getStartTime() {
32          return startTime;
33      }
34  
35      public void setStartTime(long startTime) {
36          this.startTime = startTime;
37      }
38  
39      @Override
40      public String toString() {
41          return "DbInvocation [endTime=" + endTime + ", key=" + key
42                  + ", startTime=" + startTime + ", who=" + who + "]";
43      }
44  }
