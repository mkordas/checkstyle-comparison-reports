20      public void testGetLatestDesignationFromEachUser() throws Exception {
21          MockWebCloudClient cloud = createWebCloudClient();
22          WebCloudNetworkClient spyNetworkClient = cloud.createSpyNetworkClient();
23          Mockito.when(spyNetworkClient.getIssueByHash("fad2")).thenReturn(
24                  Issue.newBuilder()
25                          .addAllEvaluations(
26                                  Arrays.asList(
27                                          Evaluation.newBuilder().setWho("user1").setDesignation("MUST_FIX")
28                                                  .setWhen(SAMPLE_DATE + 100).build(),
29                                          Evaluation.newBuilder().setWho("user2").setDesignation("I_WILL_FIX")
30                                                  .setWhen(SAMPLE_DATE + 200).build(),
31                                          Evaluation.newBuilder().setWho("user1").setDesignation("NOT_A_BUG")
32                                                  .setWhen(SAMPLE_DATE + 300).build())).build());
33          List<BugDesignation> designations = newList(cloud.getLatestDesignationFromEachUser(foundIssue));
34          assertEquals(2, designations.size());
35          assertEquals("user2", designations.get(0).getUser());
36          assertEquals("user1", designations.get(1).getUser());
37          assertEquals("NOT_A_BUG", designations.get(1).getDesignationKey());
38      }

53      public void testNormalizeHash() {
54          assertEquals("0", normalizeHash("0"));
55          assertEquals("0", normalizeHash("000000000"));
56          assertEquals("1", normalizeHash("000000000000001"));
57          assertEquals("fffffffffffffffffffffffffffffff", normalizeHash("0fffffffffffffffffffffffffffffff"));
58      }
59  
60      public void testNormalizeHashMakesLowercase() {
61          assertEquals("f", normalizeHash("F"));
62          assertEquals("fffffffffffffffffffffffffffffff", normalizeHash("0FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFF"));
63          assertEquals("ffffffffffffffffffffffffffffffff", normalizeHash("FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFF"));
64      }
