1   package edu.umd.cs.findbugs.bluej;
2   
3   import bluej.extensions.ProjectNotOpenException;
4   import bluej.extensions.event.PackageEvent;
5   import bluej.extensions.event.PackageListener;
6   
7   /**
8    * Listens for when a package opens or closes. When closes also closes the
9    * ResultsFrame of findbugs if one is open.
10   * @author Kristin Stephens
11   *
12   */
13  public class PackageClosingListener implements PackageListener {
14  
15      /**
16       * Does not do anything when package is opened.
17       */
18      public void packageOpened(PackageEvent vt) {
19      }
20  
21      /**
22       * When package is closed will also close the FindBugs results frame
23       * if it is open.
24       */
25      public void packageClosing(PackageEvent evt) {
26          try {
27              ResultsFrame frame = ResultsFrame.getInstance(evt.getPackage().getProject(), false);
28              if(frame != null)
29                  frame.setVisible(false);
30          } catch (ProjectNotOpenException e) {
31              Log.recordBug(e);
32          }
33  
34      }
35  
36  }
