19  package edu.umd.cs.findbugs.architecture;
20  
21  import java.io.File;
22  
23  import jdepend.framework.JDepend;
24  import jdepend.framework.JavaPackage;
25  import junit.framework.TestCase;
26  
27  /**
28   * Verifies the package dependencies.
29   *
30   * @author Tom\u00e1s Pollak
31   * @author Andrei Loskutov
32   */
33  public class PackageDependenciesTest extends TestCase {
34      private JDepend engine;
35  
36      public void testGui2Dependencies() {
37          String expectedNotEfferent = "edu.umd.cs.findbugs.gui2";
38  
39          assertPackageConstraint("edu.umd.cs.findbugs", expectedNotEfferent);
40          assertPackageConstraint("edu.umd.cs.findbugs.asm", expectedNotEfferent);
41          assertPackageConstraint("edu.umd.cs.findbugs.ba", expectedNotEfferent);
42          assertPackageConstraint("edu.umd.cs.findbugs.bcel", expectedNotEfferent);
43          assertPackageConstraint("edu.umd.cs.findbugs.classfile", expectedNotEfferent);
44          assertPackageConstraint("edu.umd.cs.findbugs.cloud", expectedNotEfferent);
45          assertPackageConstraint("edu.umd.cs.findbugs.detect", expectedNotEfferent);
46          assertPackageConstraint("edu.umd.cs.findbugs.graph", expectedNotEfferent);
47          assertPackageConstraint("edu.umd.cs.findbugs.io", expectedNotEfferent);
48          assertPackageConstraint("edu.umd.cs.findbugs.log", expectedNotEfferent);
49          assertPackageConstraint("edu.umd.cs.findbugs.model", expectedNotEfferent);
50          assertPackageConstraint("edu.umd.cs.findbugs.plan", expectedNotEfferent);
51          assertPackageConstraint("edu.umd.cs.findbugs.util", expectedNotEfferent);
52          assertPackageConstraint("edu.umd.cs.findbugs.visitclass", expectedNotEfferent);
53          assertPackageConstraint("edu.umd.cs.findbugs.xml", expectedNotEfferent);
54  
55      }
56  
57      @Override
58      protected void setUp() throws Exception {
59          super.setUp();
60  
61          // Get the classes root directory
62          String rootDirectory = new File(getClass().getResource("/").toURI()).getCanonicalPath();
63  
64          // Setup the JDepend analysis
65          engine = new JDepend();
66          engine.addDirectory(rootDirectory);
67          engine.analyze();
68      }
69  
70      @Override
71      protected void tearDown() throws Exception {
72          engine = null;
73  
74          super.tearDown();
75      }
76  
77      private void assertPackageConstraint(String afferent, String expectedNotEfferent) {
78          JavaPackage afferentPackage = engine.getPackage(afferent);
79          assertNotNull("Afferent package not found: " + afferent, afferentPackage);
80          JavaPackage efferentPackage = engine.getPackage(expectedNotEfferent);
81          assertNotNull("Efferent package not found: " + efferentPackage, efferentPackage);
82          assertFalse(afferentPackage.getName() + " shouldn't depend on " + efferentPackage.getName(), afferentPackage
83                  .getEfferents().contains(efferentPackage));
84      }
85  }
