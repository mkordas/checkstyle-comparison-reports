20  package edu.umd.cs.findbugs;
21  
22  import junit.framework.TestCase;
23  
24  public class OpcodeStackItemTest extends TestCase {
25      public void testMergeIntAndZero() {
26          OpcodeStack.Item intItem = new OpcodeStack.Item("I");
27          OpcodeStack.Item zeroItem = new OpcodeStack.Item("I", 0);
28          OpcodeStack.Item m1 = OpcodeStack.Item.merge(intItem, zeroItem);
29          assertNull(m1.getConstant());
30          OpcodeStack.Item m2 = OpcodeStack.Item.merge(zeroItem, intItem);
31          assertNull(m2.getConstant());
32      }
33  
34      public void testMergeTypeOnly() {
35          OpcodeStack.Item intOnly =  OpcodeStack.Item.typeOnly("I");
36          OpcodeStack.Item zeroItem = new OpcodeStack.Item("I", 0);
37  
38          OpcodeStack.Item m1 = OpcodeStack.Item.merge(intOnly, zeroItem);
39          assertEquals(0,m1.getConstant());
40          OpcodeStack.Item m2 = OpcodeStack.Item.merge(zeroItem, intOnly);
41          assertEquals(0,m2.getConstant());
42      }

44  }
