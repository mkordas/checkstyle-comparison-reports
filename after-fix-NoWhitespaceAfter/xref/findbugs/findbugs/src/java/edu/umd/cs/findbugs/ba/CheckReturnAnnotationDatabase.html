212     /**
213      * @param c
214      */
215     private void addWarningAboutSubmit(Class<? extends ExecutorService> c) {
216         addMethodAnnotation(c.getName(), "submit",
217                 "(Ljava/util/concurrent/Callable;)Ljava/util/concurrent/Future;", false,
218                 CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM_BAD_PRACTICE);
219         addMethodAnnotation(c.getName(), "submit",
220                 "(Ljava/lang/Runnable;)Ljava/util/concurrent/Future;", false,
221                 CheckReturnValueAnnotation.CHECK_RETURN_VALUE_LOW_BAD_PRACTICE);
222         addMethodAnnotation(c.getName(), "submit",
223                 "(Ljava/lang/Runnable;Ljava/lang/Object;)Ljava/util/concurrent/Future;", false,
224                 CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM_BAD_PRACTICE);
225     }

227     @Override
228     public CheckReturnValueAnnotation getResolvedAnnotation(Object o, boolean getMinimal) {
229         if (!(o instanceof XMethod)) {
230             return null;
231         }
232         XMethod m = (XMethod) o;
233         if (m.getName().startsWith("access$")) {
234             return null;
235         } else if ("<init>".equals(m.getName())) {
236             try {
237                 if (throwableClass != null && Repository.instanceOf(m.getClassName(), throwableClass)) {
238                     return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_VERY_HIGH;
239                 }
240             } catch (ClassNotFoundException e) {
241                 AnalysisContext.reportMissingClass(e);
242             }
243             if ("java.lang.Thread".equals(m.getClassName())) {
244                 return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_VERY_HIGH;
245             }
246             try {
247 
248                 if (threadClass != null && Repository.instanceOf(m.getClassName(), threadClass)) {
249                     return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_LOW;
250                 }
251             } catch (ClassNotFoundException e) {
252                 AnalysisContext.reportMissingClass(e);
253             }
254         } else if ("equals".equals(m.getName()) && "(Ljava/lang/Object;)Z".equals(m.getSignature()) && !m.isStatic()) {
255             return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM;
256         } else if (m.getSignature().endsWith(")Ljava/lang/String;")
257                 && ("java.lang.StringBuffer".equals(m.getClassName()) || "java.lang.StringBuilder".equals(m.getClassName()))) {
258             return CheckReturnValueAnnotation.CHECK_RETURN_VALUE_MEDIUM;
259         }
260         return super.getResolvedAnnotation(o, getMinimal);
261     }
