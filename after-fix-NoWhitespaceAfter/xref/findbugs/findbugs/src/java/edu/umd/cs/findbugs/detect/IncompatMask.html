20  package edu.umd.cs.findbugs.detect;
21  
22  import org.apache.bcel.classfile.Method;
23  
24  import edu.umd.cs.findbugs.BugInstance;
25  import edu.umd.cs.findbugs.BugReporter;
26  import edu.umd.cs.findbugs.BytecodeScanningDetector;
27  import edu.umd.cs.findbugs.StatelessDetector;
28  
29  /**
30   * Find comparisons involving values computed with bitwise operations whose
31   * outcomes are fixed at compile time.
32   *
33   * @author Tom Truscott <trt@unx.sas.com>
34   */
35  public class IncompatMask extends BytecodeScanningDetector implements StatelessDetector {
36      int state;
37  
38      long arg0, arg1;
39  
40      int bitop;
41  
42      boolean isLong;
43  
44      private final BugReporter bugReporter;
45  
46      public IncompatMask(BugReporter bugReporter) {
47          this.state = 0;
48          this.bugReporter = bugReporter;
49      }
50  
51      @Override
52      public void visit(Method obj) {
53          super.visit(obj);
54          this.state = 0;
55      }
56  
57      private void checkState(int expectedState) {
58          if (state == expectedState) {
59              state++;
60          } else {
61              state = 0;
62          }
63      }
64  
65      private void noteVal(long val) {
66          if (state == 0) {
67              arg0 = val;
68          } else if (state == 2) {
69              arg1 = val;
70          } else {
71              state = -1;
72          }
73          state++;
74      }
75  
76      @Override
77      public void sawInt(int val) {
78          noteVal(val);
79      }
80  
81      @Override
82      public void sawLong(long val) {
83          noteVal(val);
84      }
85  
86      static int populationCount(long i) {
87          int result = 0;
88          while (i != 0) {
89              if ((i & 1) == 1) {
90                  result++;
91              }
92              i >>>= 1;
93          }
94          return result;
95      }
96  
97      @Override
98      public void sawOpcode(int seen) {
99          // System.out.println("BIT: " + state + ": " + OPCODE_NAMES[seen]);
100 
101         switch (seen) {
102         case ICONST_M1:
103             noteVal(-1);
104             return;
105         case ICONST_0:
106             noteVal(0);
107             return;
108         case ICONST_1:
109             noteVal(1);
110             return;
111         case ICONST_2:
112             noteVal(2);
113             return;
114         case ICONST_3:
115             noteVal(3);
116             return;
117         case ICONST_4:
118             noteVal(4);
119             return;
120         case ICONST_5:
121             noteVal(5);
122             return;
123         case LCONST_0:
124             noteVal(0);
125             return;
126         case LCONST_1:
127             noteVal(1);
128             return;
129 
130         case BIPUSH:
131             return; /* will pick up value via sawInt */
132         case LDC2_W:
133             return; /* will pick up value via sawLong */
134 
135         case SIPUSH:
136             return; /* will pick up value via sawInt */
137         case LDC:
138         case LDC_W:
139             return; /* will pick up value via sawInt */
140 
141         case IAND:
142         case LAND:
143             bitop = IAND;
144             isLong = seen == LAND;
145             checkState(1);
146             return;
147         case IOR:
148         case LOR:
149             bitop = IOR;
150             isLong = seen == LOR;
151             checkState(1);
152             return;
153 
154         case LCMP:
155             if (state == 3) {
156                 isLong = true;
157                 return; /* Ignore. An 'if' opcode will follow */
158             }
159             state = 0;
160             return;
161 
162         case IFLE:
163         case IFLT:
164         case IFGT:
165         case IFGE:
166             if (state == 3 && isLong || state == 2 && !isLong) {
167                 long bits = getFlagBits(isLong, arg0);
168                 boolean highbit = !isLong && (bits & 0x80000000) != 0 || isLong && bits < 0 && bits << 1 == 0;
169                 boolean onlyLowBits = bits >>> 12 == 0;
170                 BugInstance bug;
171                 if (highbit) {
172                     bug = new BugInstance(this, "BIT_SIGNED_CHECK_HIGH_BIT", (seen == IFLE || seen == IFGT) ? HIGH_PRIORITY
173                             : NORMAL_PRIORITY);
174                 } else {
175                     bug = new BugInstance(this, "BIT_SIGNED_CHECK", onlyLowBits ? LOW_PRIORITY : NORMAL_PRIORITY);
176                 }
177                 bugReporter.reportBug(bug.addClassAndMethod(this).addSourceLine(this));
178             }
179             state = 0;
180             return;
181 
182         case IFEQ:
183         case IFNE:
184             /* special case: if arg1 is 0 it will not be pushed */
185             if (state == 2) {
186                 arg1 = 0;
187                 state = 3;
188             }
189 
190             //$FALL-THROUGH$
191         case IF_ICMPEQ:
192         case IF_ICMPNE:
193             checkState(3);
194             if (state != 4) {
195                 return;
196             }
197             break; /* the only break in this switch! gross */
198 
199         case GOTO:
200             state = -1;
201             return;
202 
203         default:
204             state = 0;
205             return;
206         }
207 
208         /* We have matched the instruction pattern, so check the args */
209         long dif;
210         String t;
211 
212         if (bitop == IOR) {
213             dif = arg0 & ~arg1;
214             t = "BIT_IOR";
215         } else if (arg0 != 0 || arg1 != 0) {
216             dif = arg1 & ~arg0;
217             t = "BIT_AND";
218         } else {
219             dif = 1;
220             t = "BIT_AND_ZZ";
221         }
222 
223         if (dif != 0) {
224             // System.out.println("Match at offset " + getPC());
225             BugInstance bug = new BugInstance(this, t, HIGH_PRIORITY).addClassAndMethod(this);
226             if (!"BIT_AND_ZZ".equals(t)) {
227                 bug.addString("0x" + Long.toHexString(arg0)).addString("0x" + Long.toHexString(arg1));
228             }
229 
230             bug.addSourceLine(this);
231             bugReporter.reportBug(bug);
232         }
233         state = 0;
234     }
235 
236     static long getFlagBits(boolean isLong, long arg0) {
237         long bits = arg0;
238         if (isLong) {
239             if (populationCount(bits) > populationCount(~bits)) {
240                 bits = ~bits;
241             }
242         } else if (populationCount(0xffffffffL & bits) > populationCount(0xffffffffL & ~bits)) {
243             bits = 0xffffffffL & ~bits;
244         }
245         return bits;
246     }
247 }
