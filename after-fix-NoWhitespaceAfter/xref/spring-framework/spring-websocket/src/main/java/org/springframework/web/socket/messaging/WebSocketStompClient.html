61  /**
62   * A STOMP over WebSocket client that connects using an implementation of
63   * {@link org.springframework.web.socket.client.WebSocketClient WebSocketClient}
64   * including {@link org.springframework.web.socket.sockjs.client.SockJsClient
65   * SockJsClient}.

67   * @author Rossen Stoyanchev
68   * @since 4.2

87  	 * Class constructor. Sets {@link #setDefaultHeartbeat} to "0,0" but will
88  	 * reset it back to the preferred "10000,10000" when a

91  	 * @param webSocketClient the WebSocket client to connect with

94  		Assert.notNull(webSocketClient, "'webSocketClient' is required.");
95  		this.webSocketClient = webSocketClient;

237 	/**
238 	 * An overloaded version of
239 	 * {@link #connect(String, StompSessionHandler, Object...)} that also accepts
240 	 * {@link WebSocketHttpHeaders} to use for the WebSocket handshake and
241 	 * {@link StompHeaders} for the STOMP CONNECT frame.
242 	 * @param url the url to connect to
243 	 * @param handshakeHeaders headers for the WebSocket handshake
244 	 * @param connectHeaders headers for the STOMP CONNECT frame
245 	 * @param handler the session handler
246 	 * @param uriVariables URI variables to expand into the URL
247 	 * @return ListenableFuture for access to the session when ready for use
248 	 */
249 	public ListenableFuture<StompSession> connect(String url, WebSocketHttpHeaders handshakeHeaders,
250 			StompHeaders connectHeaders, StompSessionHandler handler, Object... uriVariables) {
251 
252 		Assert.notNull(url, "uriTemplate must not be null");
253 		URI uri = UriComponentsBuilder.fromUriString(url).buildAndExpand(uriVariables).encode().toUri();
254 		return connect(uri, handshakeHeaders, connectHeaders, handler);
255 	}
256 
257 	/**
258 	 * An overloaded version of
259 	 * {@link #connect(String, WebSocketHttpHeaders, StompSessionHandler, Object...)}
260 	 * that accepts a fully prepared {@link java.net.URI}.
261 	 * @param url the url to connect to
262 	 * @param handshakeHeaders the headers for the WebSocket handshake
263 	 * @param connectHeaders headers for the STOMP CONNECT frame
264 	 * @param sessionHandler the STOMP session handler
265 	 * @return ListenableFuture for access to the session when ready for use
266 	 */
267 	public ListenableFuture<StompSession> connect(URI url, WebSocketHttpHeaders handshakeHeaders,
268 			StompHeaders connectHeaders, StompSessionHandler sessionHandler) {
269 
270 		Assert.notNull(url, "'uri' must not be null");
271 		ConnectionHandlingStompSession session = createSession(connectHeaders, sessionHandler);
272 		WebSocketTcpConnectionHandlerAdapter adapter = new WebSocketTcpConnectionHandlerAdapter(session);
273 		getWebSocketClient().doHandshake(adapter, handshakeHeaders, url).addCallback(adapter);
274 		return session.getSessionFuture();
275 	}
276 
277 	@Override
278 	protected StompHeaders processConnectHeaders(StompHeaders connectHeaders) {
279 		connectHeaders = super.processConnectHeaders(connectHeaders);
280 		if (connectHeaders.isHeartbeatEnabled()) {
281 			Assert.notNull(getTaskScheduler(), "TaskScheduler cannot be null if heartbeats are enabled.");
282 		}
283 		return connectHeaders;
284 	}

287 	/**
288 	 * Adapt WebSocket to the TcpConnectionHandler and TcpConnection contracts.
289 	 */
290 	private class WebSocketTcpConnectionHandlerAdapter implements ListenableFutureCallback<WebSocketSession>,
291 			WebSocketHandler, TcpConnection<byte[]> {
292 
293 		private final TcpConnectionHandler<byte[]> connectionHandler;
294 
295 		private final StompWebSocketMessageCodec codec = new StompWebSocketMessageCodec(getInboundMessageSizeLimit());
296 
297 		private volatile WebSocketSession session;
298 
299 		private volatile long lastReadTime = -1;
300 
301 		private volatile long lastWriteTime = -1;
302 
303 		private final List<ScheduledFuture<?>> inactivityTasks = new ArrayList<ScheduledFuture<?>>(2);
304 
305 
306 		public WebSocketTcpConnectionHandlerAdapter(TcpConnectionHandler<byte[]> connectionHandler) {
307 			Assert.notNull(connectionHandler);
308 			this.connectionHandler = connectionHandler;
309 		}
310 
311 
312 		// ListenableFutureCallback implementation: handshake outcome
313 
314 		@Override
315 		public void onSuccess(WebSocketSession webSocketSession) {
316 		}
317 
318 		@Override
319 		public void onFailure(Throwable ex) {
320 			this.connectionHandler.afterConnectFailure(ex);
321 		}
322 
323 
324 		// WebSocketHandler implementation
325 
326 		@Override
327 		public void afterConnectionEstablished(WebSocketSession session) {
328 			this.session = session;
329 			this.connectionHandler.afterConnected(this);
330 		}
331 
332 		@Override
333 		public void handleMessage(WebSocketSession session, WebSocketMessage<?> webSocketMessage) {
334 			this.lastReadTime = (this.lastReadTime != -1 ? System.currentTimeMillis() : -1);
335 			List<Message<byte[]>> messages;
336 			try {
337 				messages = this.codec.decode(webSocketMessage);
338 			}
339 			catch (Throwable ex) {
340 				this.connectionHandler.handleFailure(ex);
341 				return;
342 			}
343 			for (Message<byte[]> message : messages) {
344 				this.connectionHandler.handleMessage(message);
345 			}
346 		}
347 
348 		@Override
349 		public void handleTransportError(WebSocketSession session, Throwable ex) throws Exception {
350 			this.connectionHandler.handleFailure(ex);
351 		}
352 
353 		@Override
354 		public void afterConnectionClosed(WebSocketSession session, CloseStatus closeStatus) throws Exception {
355 			cancelInactivityTasks();
356 			this.connectionHandler.afterConnectionClosed();
357 		}
358 
359 		private void cancelInactivityTasks() {
360 			for (ScheduledFuture<?> task : this.inactivityTasks) {
361 				try {
362 					task.cancel(true);
363 				}
364 				catch (Throwable ex) {
365 					// Ignore
366 				}
367 			}
368 			this.lastReadTime = -1;
369 			this.lastWriteTime = -1;
370 			this.inactivityTasks.clear();
371 		}
372 
373 		@Override
374 		public boolean supportsPartialMessages() {
375 			return false;
376 		}
377 
378 
379 		// TcpConnection implementation
380 
381 		@Override
382 		public ListenableFuture<Void> send(Message<byte[]> message) {
383 			updateLastWriteTime();
384 			SettableListenableFuture<Void> future = new SettableListenableFuture<Void>();
385 			try {
386 				this.session.sendMessage(this.codec.encode(message, this.session.getClass()));
387 				future.set(null);
388 			}
389 			catch (Throwable ex) {
390 				future.setException(ex);
391 			}
392 			finally {
393 				updateLastWriteTime();
394 			}
395 			return future;
396 		}
397 
398 		private void updateLastWriteTime() {
399 			this.lastWriteTime = (this.lastWriteTime != -1 ? System.currentTimeMillis() : -1);
400 		}
401 
402 		@Override
403 		public void onReadInactivity(final Runnable runnable, final long duration) {
404 			Assert.notNull(getTaskScheduler(), "No scheduler configured.");
405 			this.lastReadTime = System.currentTimeMillis();
406 			this.inactivityTasks.add(getTaskScheduler().scheduleWithFixedDelay(new Runnable() {
407 				@Override
408 				public void run() {
409 					if (System.currentTimeMillis() - lastReadTime > duration) {
410 						try {
411 							runnable.run();
412 						}
413 						catch (Throwable ex) {
414 							if (logger.isDebugEnabled()) {
415 								logger.debug("ReadInactivityTask failure", ex);
416 							}
417 						}
418 					}
419 				}
420 			}, duration / 2));
421 		}
422 
423 		@Override
424 		public void onWriteInactivity(final Runnable runnable, final long duration) {
425 			Assert.notNull(getTaskScheduler(), "No scheduler configured.");
426 			this.lastWriteTime = System.currentTimeMillis();
427 			this.inactivityTasks.add(getTaskScheduler().scheduleWithFixedDelay(new Runnable() {
428 				@Override
429 				public void run() {
430 					if (System.currentTimeMillis() - lastWriteTime > duration) {
431 						try {
432 							runnable.run();
433 						}
434 						catch (Throwable ex) {
435 							if (logger.isDebugEnabled()) {
436 								logger.debug("WriteInactivityTask failure", ex);
437 							}
438 						}
439 					}
440 				}
441 			}, duration / 2));
442 		}
443 
444 		@Override
445 		public void close() {
446 			try {
447 				this.session.close();
448 			}
449 			catch (IOException ex) {
450 				if (logger.isDebugEnabled()) {
451 					logger.debug("Failed to close session: " + this.session.getId(), ex);
452 				}
453 			}
454 		}
455 	}

458 	/**
459 	 * Encode and decode STOMP WebSocket messages.
460 	 */
461 	private static class StompWebSocketMessageCodec {
462 
463 		private static final StompEncoder ENCODER = new StompEncoder();
464 
465 		private static final StompDecoder DECODER = new StompDecoder();
466 
467 		private final BufferingStompDecoder bufferingDecoder;
468 
469 
470 		public StompWebSocketMessageCodec(int messageSizeLimit) {
471 			this.bufferingDecoder = new BufferingStompDecoder(DECODER, messageSizeLimit);
472 		}
473 
474 
475 		public List<Message<byte[]>> decode(WebSocketMessage<?> webSocketMessage) {
476 			List<Message<byte[]>> result = Collections.<Message<byte[]>>emptyList();
477 			ByteBuffer byteBuffer;
478 			if (webSocketMessage instanceof TextMessage) {
479 				byteBuffer = ByteBuffer.wrap(((TextMessage) webSocketMessage).asBytes());
480 			}
481 			else if (webSocketMessage instanceof BinaryMessage) {
482 				byteBuffer = ((BinaryMessage) webSocketMessage).getPayload();
483 			}
484 			else {
485 				return result;
486 			}
487 			result = this.bufferingDecoder.decode(byteBuffer);
488 			if (result.isEmpty()) {
489 				if (logger.isTraceEnabled()) {
490 					logger.trace("Incomplete STOMP frame content received, bufferSize=" +
491 							this.bufferingDecoder.getBufferSize() + ", bufferSizeLimit=" +
492 							this.bufferingDecoder.getBufferSizeLimit() + ".");
493 				}
494 			}
495 			return result;
496 		}
497 
498 		public WebSocketMessage<?> encode(Message<byte[]> message, Class<? extends WebSocketSession> sessionType) {
499 			StompHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);
500 			Assert.notNull(accessor);
501 			byte[] payload = message.getPayload();
502 			byte[] bytes = ENCODER.encode(accessor.getMessageHeaders(), payload);
503 
504 			boolean useBinary = (payload.length > 0  &&
505 					!(SockJsSession.class.isAssignableFrom(sessionType)) &&
506 					MimeTypeUtils.APPLICATION_OCTET_STREAM.isCompatibleWith(accessor.getContentType()));
507 
508 			return (useBinary ? new BinaryMessage(bytes) : new TextMessage(bytes));
509 		}
510 	}
