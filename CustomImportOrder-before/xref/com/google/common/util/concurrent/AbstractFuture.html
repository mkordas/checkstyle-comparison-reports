41  /**
42   * An abstract implementation of the {@link ListenableFuture} interface. This
43   * class is preferable to {@link java.util.concurrent.FutureTask} for two
44   * reasons: It implements {@code ListenableFuture}, and it does not implement
45   * {@code Runnable}. (If you want a {@code Runnable} implementation of {@code
46   * ListenableFuture}, create a {@link ListenableFutureTask}, or submit your
47   * tasks to a {@link ListeningExecutorService}.)
48   *
49   * <p>This class implements all methods in {@code ListenableFuture}.
50   * Subclasses should provide a way to set the result of the computation through
51   * the protected methods {@link #set(Object)},
52   * {@link #setFuture(ListenableFuture)} and {@link #setException(Throwable)}.
53   * Subclasses may also override {@link #interruptTask()}, which will be invoked
54   * automatically if a call to {@link #cancel(boolean) cancel(true)} succeeds in
55   * canceling the future.
56   *
57   * <p>Subclasses should rarely override other methods. Instead, prefer designs
58   * that derive a new {@code Future} to return to clients, typically using
59   * methods like {@link Futures#transform(ListenableFuture, AsyncFunction)
60   * Futures.transform} and {@link Futures#withFallback(ListenableFuture,
61   * FutureFallback) Futures.withFallback}.
62   *
63   * @author Sven Mawson
64   * @author Luke Sandberg
65   * @since 1.0
66   */
67  public abstract class AbstractFuture<V> implements ListenableFuture<V> {
68    // Logger to log exceptions caught when running listeners.
69    private static final Logger log = Logger.getLogger(AbstractFuture.class.getName());

108   /**
109    * Waiter links form a Treiber stack, in the {@link #waiters} field.
110    */
111   private static final class Waiter {
112     static final Waiter TOMBSTONE = new Waiter(false /* ignored param */);
113 
114     @Nullable volatile Thread thread;
115     @Nullable volatile Waiter next;
116 
117     // Constructor for the TOMBSTONE, avoids use of ATOMIC_HELPER in case this class is loaded
118     // before the ATOMIC_HELPER.  Apparently this is possible on some android platforms.
119     Waiter(boolean ignored) {}
120 
121     Waiter() {
122       // avoid volatile write, write is made visible by subsequent CAS on waiters field
123       ATOMIC_HELPER.putThread(this, Thread.currentThread());
124     }
125 
126     // non-volatile write to the next field. Should be made visible by subsequent CAS on waiters
127     // field.
128     void setNext(Waiter next) {
129       ATOMIC_HELPER.putNext(this, next);
130     }
131 
132     void unpark() {
133       // This is racy with removeWaiter.  The consequence of the race is that we may spuriously
134       // call unpark even though the thread has already removed itself from the list.  But even if
135       // we did use a CAS, that race would still exist (it would just be ever so slightly smaller).
136       Thread w = thread;
137       if (w != null) {
138         thread = null;
139         LockSupport.unpark(w);
140       }
141     }
142   }

144   /**
145    * Marks the given node as 'deleted' (null waiter) and then scans the list to unlink all deleted
146    * nodes.  This is an O(n) operation in the common case (and O(n^2) in the worst), but we are
147    * saved by two things.
148    * <ul>
149    *   <li>This is only called when a waiting thread times out or is interrupted.  Both of which
150    *       should be rare.
151    *   <li>The waiters list should be very short.
152    * </ul>
153    */
154   private void removeWaiter(Waiter node) {
155     node.thread = null;  // mark as 'deleted'
156     restart: while (true) {
157       Waiter pred = null;
158       Waiter curr = waiters;
159       if (curr == Waiter.TOMBSTONE) {
160         return;  // give up if someone is calling complete
161       }
162       Waiter succ;
163       while (curr != null) {
164         succ = curr.next;
165         if (curr.thread != null) {  // we aren't unlinking this node, update pred.
166           pred = curr;
167         } else if (pred != null) { // We are unlinking this node and it has a predecessor.
168           pred.next = succ;
169           if (pred.thread == null) {  // We raced with another node that unlinked pred. Restart.
170             continue restart;
171           }
172         } else if (!ATOMIC_HELPER.casWaiters(this, curr, succ)) {  // We are unlinking head
173           continue restart;  // We raced with an add or complete
174         }
175         curr = succ;
176       }
177       break;
178     }
179   }

276   // Gets and Timed Gets
277   //
278   // * Be responsive to interruption
279   // * Don't create Waiter nodes if you aren't going to park, this helps reduce contention on the
280   //   waiters field.
281   // * Future completion is defined by when #value becomes non-null/non SetFuture
282   // * Future completion can be observed if the waiters field contains a TOMBSTONE
283 
284   // Timed Get
285   // There are a few design constraints to consider
286   // * We want to be responsive to small timeouts, unpark() has non trivial latency overheads (I
287   //   have observed 12 micros on 64 bit linux systems to wake up a parked thread).  So if the
288   //   timeout is small we shouldn't park().  This needs to be traded off with the cpu overhead of
289   //   spinning, so we use SPIN_THRESHOLD_NANOS which is what AbstractQueuedSynchronizer uses for
290   //   similar purposes.
291   // * We want to behave reasonably for timeouts of 0
292   // * We are more responsive to completion than timeouts.  This is because parkNanos depends on
293   //   system scheduling and as such we could either miss our deadline, or unpark() could be delayed
294   //   so that it looks like we timed out even though we didn't.  For comparison FutureTask respects
295   //   completion preferably and AQS is non-deterministic (depends on where in the queue the waiter
296   //   is).  If we wanted to be strict about it, we could store the unpark() time in the Waiter
297   //   node and we could use that to make a decision about whether or not we timed out prior to
298   //   being unparked.
299 
300   /**
301    * {@inheritDoc}
302    *
303    * <p>The default {@link AbstractFuture} implementation throws {@code
304    * InterruptedException} if the current thread is interrupted before or during
305    * the call, even if the value is already available.
306    *
307    * @throws InterruptedException if the current thread was interrupted before
308    *     or during the call (optional but recommended).
309    * @throws CancellationException {@inheritDoc}
310    */
311   @Override
312   public V get(long timeout, TimeUnit unit)
313       throws InterruptedException, TimeoutException, ExecutionException {
314     // NOTE: if timeout < 0, remainingNanos will be < 0 and we will fall into the while(true) loop
315     // at the bottom and throw a timeoutexception.
316     long remainingNanos = unit.toNanos(timeout); // we rely on the implicit null check on unit.
317     if (Thread.interrupted()) {
318       throw new InterruptedException();
319     }
320     Object localValue = value;
321     if (localValue != null & !(localValue instanceof AbstractFuture.SetFuture)) {
322       return getDoneValue(localValue);
323     }
324     // we delay calling nanoTime until we know we will need to either park or spin
325     final long endNanos = remainingNanos > 0 ? System.nanoTime() + remainingNanos : 0;
326     // This may look like a 'while' but it is really an 'if' that we can break <label> out of.
327     long_wait_loop: while (remainingNanos > SPIN_THRESHOLD_NANOS) {
328       Waiter oldHead = waiters;
329       if (oldHead != Waiter.TOMBSTONE) {
330         Waiter node = new Waiter();
331         do {
332           node.setNext(oldHead);
333           if (ATOMIC_HELPER.casWaiters(this, oldHead, node)) {
334             while (true) {
335               LockSupport.parkNanos(this, remainingNanos);
336               // Check interruption first, if we woke up due to interruption we need to honor that.
337               if (Thread.interrupted()) {
338                 removeWaiter(node);
339                 throw new InterruptedException();
340               }
341 
342               // Otherwise re-read and check doneness.  If we loop then it must have been a spurious
343               // wakeup
344               localValue = value;
345               if (localValue != null & !(localValue instanceof AbstractFuture.SetFuture)) {
346                 return getDoneValue(localValue);
347               }
348 
349               // timed out?
350               remainingNanos = endNanos - System.nanoTime();
351               if (remainingNanos < SPIN_THRESHOLD_NANOS) {
352                 // Remove the waiter, one way or another we are done parking this thread.
353                 removeWaiter(node);
354                 break long_wait_loop;  // jump down to the busy wait loop
355               }
356             }
357           }
358           oldHead = waiters;  // re-read and loop.
359         } while (oldHead != Waiter.TOMBSTONE);
360       }
361       // re-read value, if we get here then we must have observed a TOMBSTONE while trying to add a
362       // waiter.
363       return getDoneValue(value);
364     }
365     // If we get here then we have remainingNanos < SPIN_THRESHOLD_NANOS and there is no node on the
366     // waiters list
367     while (remainingNanos > 0) {
368       localValue = value;
369       if (localValue != null & !(localValue instanceof AbstractFuture.SetFuture)) {
370         return getDoneValue(localValue);
371       }
372       if (Thread.interrupted()) {
373         throw new InterruptedException();
374       }
375       remainingNanos = endNanos - System.nanoTime();
376     }
377     throw new TimeoutException();
378   }

384   /**
385    * {@inheritDoc}
386    *
387    * <p>The default {@link AbstractFuture} implementation throws {@code
388    * InterruptedException} if the current thread is interrupted before or during
389    * the call, even if the value is already available.
390    *
391    * @throws InterruptedException if the current thread was interrupted before
392    *     or during the call (optional but recommended).
393    * @throws CancellationException {@inheritDoc}
394    */
395   @Override public V get() throws InterruptedException, ExecutionException {
396     if (Thread.interrupted()) {
397       throw new InterruptedException();
398     }
399     Object localValue = value;
400     if (localValue != null & !(localValue instanceof AbstractFuture.SetFuture)) {
401       return getDoneValue(localValue);
402     }
403     Waiter oldHead = waiters;
404     if (oldHead != Waiter.TOMBSTONE) {
405       Waiter node = new Waiter();
406       do {
407         node.setNext(oldHead);
408         if (ATOMIC_HELPER.casWaiters(this, oldHead, node)) {
409           // we are on the stack, now wait for completion.
410           while (true) {
411             LockSupport.park(this);
412             // Check interruption first, if we woke up due to interruption we need to honor that.
413             if (Thread.interrupted()) {
414               removeWaiter(node);
415               throw new InterruptedException();
416             }
417             // Otherwise re-read and check doneness.  If we loop then it must have been a spurious
418             // wakeup
419             localValue = value;
420             if (localValue != null & !(localValue instanceof AbstractFuture.SetFuture)) {
421               return getDoneValue(localValue);
422             }
423           }
424         }
425         oldHead = waiters;  // re-read and loop.
426       } while (oldHead != Waiter.TOMBSTONE);
427     }
428     // re-read value, if we get here then we must have observed a TOMBSTONE while trying to add a
429     // waiter.
430     return getDoneValue(value);
431   }

433   /**
434    * Unboxes {@code obj}.  Assumes that obj is not {@code null} or a {@link SetFuture}.
435    */
436   private V getDoneValue(Object obj) throws ExecutionException {
437     // While this seems like it might be too branch-y, simple benchmarking proves it to be
438     // unmeasurable (comparing done AbstractFutures with immediateFuture)
439     if (obj instanceof Cancellation) {
440       throw cancellationExceptionWithCause("Task was cancelled.", ((Cancellation) obj).cause);
441     } else if (obj instanceof Failure) {
442       throw new ExecutionException(((Failure) obj).exception);
443     } else if (obj == NULL) {
444       return null;
445     } else {
446       @SuppressWarnings("unchecked")  // this is the only other option
447       V asV = (V) obj;
448       return asV;
449     }
450   }

464   /**
465    * {@inheritDoc}
466    *
467    * <p>If a cancellation attempt succeeds on a {@code Future} that had previously been {@linkplain
468    * #setFuture asynchronously set}, then the cancellation will also be propagated to the delegate
469    * {@code Future} that was supplied in the {@code setFuture} call.
470    */
471   @Override
472   public boolean cancel(boolean mayInterruptIfRunning) {
473     Object localValue = value;
474     if (localValue == null | localValue instanceof AbstractFuture.SetFuture) {
475       // Try to delay allocating the exception.  At this point we may still lose the CAS, but it is
476       // certainly less likely.
477       // TODO(user): this exception actually makes cancellation significantly more expensive :(
478       // I wonder if we should consider removing it or providing a mechanism to not do it.
479       Object valueToSet = new Cancellation(mayInterruptIfRunning,
480           new CancellationException("Future.cancel() was called."));
481       do {
482         if (ATOMIC_HELPER.casValue(this, localValue, valueToSet)) {
483           // We call interuptTask before calling complete(), first which is consistent with
484           // FutureTask
485           if (mayInterruptIfRunning) {
486             interruptTask();
487           }
488           complete();
489           if (localValue instanceof AbstractFuture.SetFuture) {
490             // propagate cancellation to the future set in setfuture, this is racy, and we don't
491             // care if we are successful or not.
492             ((AbstractFuture<?>.SetFuture) localValue).future.cancel(mayInterruptIfRunning);
493           }
494           return true;
495         }
496         // obj changed, reread
497         localValue = value;
498         // obj cannot be null at this point, because value can only change from null to non-null. So
499         // if value changed (and it did since we lost the CAS), then it cannot be null.
500       } while (localValue instanceof AbstractFuture.SetFuture);
501     }
502     return false;
503   }

533   @Override
534   public void addListener(Runnable listener, Executor executor) {
535     checkNotNull(listener, "Runnable was null.");
536     checkNotNull(executor, "Executor was null.");
537     Listener oldHead = listeners;
538     if (oldHead != Listener.TOMBSTONE) {
539       Listener newNode = new Listener(listener, executor);
540       do {
541         newNode.next = oldHead;
542         if (ATOMIC_HELPER.casListeners(this, oldHead, newNode)) {
543           return;
544         }
545         oldHead = listeners;  // re-read
546       } while (oldHead != Listener.TOMBSTONE);
547     }
548     // If we get here then the Listener TOMBSTONE was set, which means the future is done, call
549     // the listener.
550     executeListener(listener, executor);
551   }

553   /**
554    * Sets the result of this {@code Future} unless this {@code Future} has previously been cancelled
555    * or set (including {@linkplain #setFuture set asynchronously}). When a call to this method
556    * returns, the {@code Future} is guaranteed to be {@linkplain #isDone done} <b>only if</b> the
557    * call was accepted (in which case it returns {@code true}). If it returns {@code false}, the
558    * {@code Future} may have previously been set asynchronously, in which case its result may not be
559    * known yet. That result, though not yet known, cannot by overridden by a call to a {@code set*}
560    * method, only by a call to {@link #cancel}.
561    *
562    * @param value the value to be used as the result
563    * @return true if the attempt was accepted, completing the {@code Future}
564    */
565   protected boolean set(@Nullable V value) {
566     Object valueToSet = value == null ? NULL : value;
567     if (ATOMIC_HELPER.casValue(this, null, valueToSet)) {
568       complete();
569       return true;
570     }
571     return false;
572   }

595   /**
596    * Asynchronously sets the result of this {@code Future} unless this {@code Future} has previously
597    * been cancelled or set (including {@linkplain #setFuture set asynchronously}). If a call to this
598    * method is accepted, this {@code Future} will complete when the supplied {@code Future}
599    * completes or when this {@code Future} is cancelled (at which point cancellation will also be
600    * propagated to the supplied {@code Future}).
601    *
602    * <p>If the supplied future is {@linkplain #isDone done} when this method is called and the call
603    * is accepted, then this future is guaranteed to have been completed with the supplied future.
604    * If the supplied future is not done and the call is accepted, then the future will be set
605    * asynchronously. Note that such a result, though not yet known, cannot by overridden by a call
606    * to a {@code set*} method, only by a call to {@link #cancel}.
607    *
608    * @param future the future to delegate to
609    * @return true if the attempt was accepted, indicating that the {@code Future} was not previously
610    *     cancelled or set.
611    * @since 19.0
612    */
613   @Beta protected boolean setFuture(ListenableFuture<? extends V> future) {
614     checkNotNull(future);
615     Object localValue = value;
616     if (localValue == null) {
617       if (future.isDone()) {
618         return completeWithFuture(future, null);
619       }
620       SetFuture valueToSet = new SetFuture(future);
621       if (ATOMIC_HELPER.casValue(this, null, valueToSet)) {
622         // the listener is responsible for calling completeWithFuture, directExecutor is appropriate
623         // since all we are doing is unpacking a completed future which should be fast.
624         try {
625           future.addListener(valueToSet, directExecutor());
626         } catch (Throwable t) {
627           // addListener has thrown an exception!  SetFuture.run can't throw any exceptions so this
628           // must have been caused by addListener itself.  The most likely explanation is a
629           // misconfigured mock.  Try to switch to Failure.
630           Failure failure;
631           try {
632             failure = new Failure(t);
633           } catch (Throwable oomMostLikely) {
634             failure = Failure.FALLBACK_INSTANCE;
635           }
636           // Note: The only way this CAS could fail is if cancel() has raced with us. That is ok.
637           ATOMIC_HELPER.casValue(this, valueToSet, failure);
638         }
639         return true;
640       }
641       localValue = value;  // we lost the cas, fall through and maybe cancel
642     }
643     // The future has already been set to something.  If it is cancellation we should cancel the
644     // incoming future.
645     if (localValue instanceof Cancellation) {
646       // we don't care if it fails, this is best-effort.
647       future.cancel(((Cancellation) localValue).wasInterrupted);
648     }
649     return false;
650   }

652   /**
653    * Called when a future passed via setFuture has completed.
654    *
655    * @param future the done future to complete this future with.
656    * @param expected the expected value of the {@link #value} field.
657    */
658   private boolean completeWithFuture(ListenableFuture<? extends V> future, Object expected) {
659     // TODO(user): Specialize this method in the case that future is an AbstractFuture.  Then
660     // instead of calling .get() and catching exception we can just copy the value field which
661     // should be much cheaper (a single cast and a volatile read, instead of at least 2 reads,
662     // dealing with InterruptedException and possibly throwing/catching exceptions).  The issue is
663     // that some subclasses override .get() and may expect/require it to be called and this would
664     // break those assumptions. Possible ideas for managing this:
665     // 1. limit the optimization to a trusted set of subclasses (subclasses in this package?
666     //    via a package private interface?)
667     // 2. entirely change the subclassing interface e.g. make .get() final. Then users who want to
668     //    do fancy things in .get() will need to use ForwardingFuture.
669     Object valueToSet;
670     try {
671       V v = Uninterruptibles.getUninterruptibly(future);
672       valueToSet = v == null ? NULL : v;
673     } catch (ExecutionException exception) {
674       valueToSet = new Failure(exception.getCause());
675     } catch (CancellationException cancellation) {
676       valueToSet = new Cancellation(false, cancellation);
677     } catch (Throwable t) {
678       valueToSet = new Failure(t);
679     }
680     // The only way this can fail is if we raced with another thread calling cancel(). If we lost
681     // that race then there is nothing to do.
682     if (ATOMIC_HELPER.casValue(AbstractFuture.this, expected, valueToSet)) {
683       complete();
684       return true;
685     }
686     return false;
687   }

689   /** Unblocks all threads and runs all listeners. */
690   private void complete() {
691     for (Waiter currentWaiter = clearWaiters();
692         currentWaiter != null;
693         currentWaiter = currentWaiter.next) {
694       currentWaiter.unpark();
695     }
696     // We need to reverse the list to handle buggy listeners that depend on ordering.
697     Listener currentListener = clearListeners();
698     Listener reversedList = null;
699     while (currentListener != null) {
700       Listener tmp = currentListener;
701       currentListener = currentListener.next;
702       tmp.next = reversedList;
703       reversedList = tmp;
704     }
705     for (; reversedList != null; reversedList = reversedList.next) {
706       executeListener(reversedList.task, reversedList.executor);
707     }
708   }
709 
710   /** Clears the {@link #waiters} list and returns the most recently added value. */
711   private Waiter clearWaiters() {
712     Waiter head;
713     do {
714       head = waiters;
715     } while (!ATOMIC_HELPER.casWaiters(this, head, Waiter.TOMBSTONE));
716     return head;
717   }
718 
719   /** Clears the {@link #listeners} list and returns the most recently added value. */
720   private Listener clearListeners() {
721     Listener head;
722     do {
723       head = listeners;
724     } while (!ATOMIC_HELPER.casListeners(this, head, Listener.TOMBSTONE));
725     return head;
726   }

728   /**
729    * Submits the given runnable to the given {@link Executor} catching and logging all
730    * {@linkplain RuntimeException runtime exceptions} thrown by the executor.
731    */
732   private static void executeListener(Runnable runnable, Executor executor) {
733     try {
734       executor.execute(runnable);
735     } catch (RuntimeException e) {
736       // Log it and keep going, bad runnable and/or executor.  Don't
737       // punish the other runnables if we're given a bad one.  We only
738       // catch RuntimeException because we want Errors to propagate up.
739       log.log(Level.SEVERE, "RuntimeException while executing runnable "
740           + runnable + " with executor " + executor, e);
741     }
742   }

747     exception.initCause(cause);

768   /**
769    * {@link AtomicHelper} based on {@link sun.misc.Unsafe}.
770    *
771    * <p>Static initialization of this class will fail if the {@link sun.misc.Unsafe} object cannot
772    * be accessed.
773    */
774   private static final class UnsafeAtomicHelper extends AtomicHelper {
775     static final sun.misc.Unsafe UNSAFE;
776     static final long LISTENERS_OFFSET;
777     static final long WAITERS_OFFSET;
778     static final long VALUE_OFFSET;
779     static final long WAITER_THREAD_OFFSET;
780     static final long WAITER_NEXT_OFFSET;
781 
782     static {
783       sun.misc.Unsafe unsafe = null;
784       try {
785         unsafe = sun.misc.Unsafe.getUnsafe();
786       } catch (SecurityException tryReflectionInstead) {
787         try {
788           unsafe = AccessController.doPrivileged(
789               new PrivilegedExceptionAction<sun.misc.Unsafe>() {
790                 @Override public sun.misc.Unsafe run() throws Exception {
791                   Class<sun.misc.Unsafe> k = sun.misc.Unsafe.class;
792                   for (java.lang.reflect.Field f : k.getDeclaredFields()) {
793                     f.setAccessible(true);
794                     Object x = f.get(null);
795                     if (k.isInstance(x)) {
796                       return k.cast(x);
797                     }
798                   }
799                   throw new NoSuchFieldError("the Unsafe");
800                 }
801               });
802         } catch (PrivilegedActionException e) {
803           throw new RuntimeException("Could not initialize intrinsics", e.getCause());
804         }
805       }
806       try {
807         Class<?> abstractFuture = AbstractFuture.class;
808         WAITERS_OFFSET = unsafe.objectFieldOffset(abstractFuture.getDeclaredField("waiters"));
809         LISTENERS_OFFSET = unsafe.objectFieldOffset(abstractFuture.getDeclaredField("listeners"));
810         VALUE_OFFSET = unsafe.objectFieldOffset(abstractFuture.getDeclaredField("value"));
811         WAITER_THREAD_OFFSET = unsafe.objectFieldOffset(Waiter.class.getDeclaredField("thread"));
812         WAITER_NEXT_OFFSET = unsafe.objectFieldOffset(Waiter.class.getDeclaredField("next"));
813         UNSAFE = unsafe;
814       } catch (Exception e) {
815         throw Throwables.propagate(e);
816       }
817     }
818 
819     @Override void putThread(Waiter waiter, Thread thread) {
820       UNSAFE.putObject(waiter, WAITER_THREAD_OFFSET, thread);
821     }
822 
823     @Override void putNext(Waiter waiter, Waiter next) {
824       UNSAFE.putObject(waiter, WAITER_NEXT_OFFSET, next);
825     }
826 
827     /** Performs a CAS operation on the {@link #waiters} field. */
828     @Override boolean casWaiters(AbstractFuture future, Waiter curr, Waiter next) {
829       return UNSAFE.compareAndSwapObject(future, WAITERS_OFFSET, curr, next);
830     }
831 
832     /** Performs a CAS operation on the {@link #listeners} field. */
833     @Override boolean casListeners(AbstractFuture future, Listener curr, Listener next) {
834       return UNSAFE.compareAndSwapObject(future, LISTENERS_OFFSET, curr, next);
835     }
836 
837     /** Performs a CAS operation on the {@link #value} field. */
838     @Override boolean casValue(AbstractFuture future, Object expected, Object v) {
839       return UNSAFE.compareAndSwapObject(future, VALUE_OFFSET, expected, v);
840     }
841   }
